function isAdjacent(a, b) {
  if (a === b) return false;
  const r1 = Math.floor(a / 3), c1 = a % 3;
  const r2 = Math.floor(b / 3), c2 = b % 3;
  return Math.abs(r1 - r2) <= 1 && Math.abs(c1 - c2) <= 1;
}
